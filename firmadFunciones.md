## Firmas de funciones

```c#
  // 1. Función sin parámetros ni retorno
static void Saludar();

// 2. Función con un parámetro y sin retorno
static void SaludarPersona(string nombre);

// 3. Función con retorno sin parámetros
static int ObtenerNumero();

// 4. Función con parámetros y retorno
static int Sumar(int a, int b);

// 5. Función que devuelve el mayor de dos números
static int Mayor(int a, int b);

// 6. Función que verifica si un número es par
static bool EsPar(int numero);

// 7. Función que calcula el cuadrado de un número
static int Cuadrado(int numero);

// 8. Función que concatena dos cadenas de texto
static string Concatenar(string texto1, string texto2);

// 9. Función con params (cantidad variable de argumentos)
static int SumarTodos(params int[] numeros);

// 10. Función que devuelve el menor de tres números
static int Menor(int a, int b, int c);

// 11. Función que convierte un número a texto
static string NumeroATexto(int numero);

// 12. Función que calcula el área de un rectángulo
static int AreaRectangulo(int baseRect, int altura);

// 13. Función que verifica si un número es positivo
static bool EsPositivo(int numero);

// 14. Función que devuelve el primer carácter de una cadena
static char PrimerCaracter(string texto);

// 15. Función que genera un saludo personalizado
static string ObtenerSaludo(string nombre);

// 16. Función que multiplica tres números
static int MultiplicarTres(int a, int b, int c);

// 17. Función con sobrecarga (multiplicación)
static int Multiplicar(int a, int b);
static double Multiplicar(double a, double b);

// 18. Función que devuelve el último elemento de una lista
static int UltimoElemento(List<int> numeros);

// 19. Función que dice si una cadena contiene otra palabra
static bool ContienePalabra(string frase, string palabra);

// 20. Función que cuenta los caracteres de una cadena
static int ContarCaracteres(string texto);

```
